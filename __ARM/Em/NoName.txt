function zz (V : OleVariant)  : TP;
var P : pointer;
begin
  P := VarArrayLock(V);
  Move(P^, Result, sizeOf(TP));
  VarArrayUnlock(V);
end;

function rr (A : TP) : Variant;
var P : pointer;
begin
  Result := VarArrayCreate([0, sizeOf(TP)], varByte);
  P := VarArrayLock(Result);
  Move(A, P^, sizeOf(A));
  VarArrayUnlock(Result);
end;
